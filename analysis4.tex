\section{–ê–ª–≥–µ–±—Ä—ã –∏ \texorpdfstring{$\sigma$}{œÉ}-–∞–ª–≥–µ–±—Ä—ã –º–Ω–æ–∂–µ—Å—Ç–≤.}

	\begin{de}
		–ü—É—Å—Ç—å $X$~--- –Ω–µ–∫–æ—Ç–æ—Ä–æ–µ –º–Ω–æ–∂–µ—Å—Ç–≤–æ. –¢–æ–≥–¥–∞ $\mc{A} \subset 2^{X}$ –Ω–∞–∑—ã–≤–∞–µ—Ç—Å—è \ti{–∞–ª–≥–µ–±—Ä–æ–π,} –µ—Å–ª–∏ –≤—ã–ø–æ–ª–Ω—è—é—Ç—Å—è —Å–ª–µ–¥—É—é—â–∏–µ —É—Å–ª–æ–≤–∏—è:
		\begin{enumerate}
			\item $\varnothing, \, X \in \mc{A},$
			\item $A, \, B \in \mc{A} \so A \cup B, \, A \cap B, \, A \setminus B \in \mc{A}.$
		\end{enumerate} 
	\end{de}

	\begin{exc}
		–ü—É—Å—Ç—å $\mc{A} \subset 2^X$~--- –∞–ª–≥–µ–±—Ä–∞, $|\mc{A}| < \infty$. –¢–æ–≥–¥–∞ $|\mc{A}| = 2^n$ –¥–ª—è –Ω–µ–∫–æ—Ç–æ—Ä–æ–≥–æ $n$.
		\begin{proof}
			–¢–∞–∫ –∫–∞–∫ $X \in \mc{A},$ –∫–∞–∂–¥—ã–π —ç–ª–µ–º–µ–Ω—Ç $X$ —Å–æ–¥–µ—Ä–∂–∏—Ç—Å—è –∫–∞–∫ –º–∏–Ω–∏–º—É–º –≤ –æ–¥–Ω–æ–º —ç–ª–µ–º–µ–Ω—Ç–µ $\mc{A}$. –ü—É—Å—Ç—å $A(x)$~--- –ø–µ—Ä–µ—Å–µ—á–µ–Ω–∏–µ –≤—Å–µ—Ö –º–Ω–æ–∂–µ—Å—Ç–≤ –∏–∑ $\mc{A},$ —Å–æ–¥–µ—Ä–∂–∞—â–∏—Ö $x$. –ü–æ–Ω—è—Ç–Ω–æ, —á—Ç–æ $A(x)$ –Ω–µ–ø—É—Å—Ç–æ, —Ç.–∫. $x \in A(x)$. –†–∞–∑–æ–±—å—ë–º –¥–∞–ª—å–Ω–µ–π—à–µ–µ –¥–æ–∫–∞–∑–∞—Ç–µ–ª—å—Å—Ç–≤–æ –Ω–∞ –Ω–µ—Å–∫–æ–ª—å–∫–æ –ø—É–Ω–∫—Ç–æ–≤:
			\begin{enumerate}
				\item –ú—ã –æ–ø—Ä–µ–¥–µ–ª–∏–ª–∏ $A(x),$ –∫–∞–∫ –Ω–∞–∏–º–µ–Ω—å—à–µ–µ –ø–æ –≤–∫–ª—é—á–µ–Ω–∏—é –º–Ω–æ–∂–µ—Å—Ç–≤–æ, —É–¥–æ–≤–ª–µ—Ç–≤–æ—Ä—è—é—â–µ–µ –Ω–µ–∫–æ—Ç–æ—Ä–æ–º—É —Å–≤–æ–π—Å—Ç–≤—É. –ü–æ—ç—Ç–æ–º—É —É –Ω–µ–≥–æ –µ—Å—Ç—å —ç–∫–≤–∏–≤–∞–ª–µ–Ω—Ç–Ω–æ–µ –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ: $A(x)$~--- —Ç–∞–∫–æ–µ –º–Ω–æ–∂–µ—Å—Ç–≤–æ, —á—Ç–æ –µ—Å–ª–∏ $x \in B \in \mc{A},$ —Ç–æ $A(x) \subset B$ \footnote{–ó–∞–º–µ—Ç–∏–º, —á—Ç–æ –º—ã —Å—É—â–µ—Å—Ç–≤–µ–Ω–Ω–æ –∏—Å–ø–ª—å–∑—É–µ–º –∫–æ–Ω–µ—á–Ω–æ—Å—Ç—å $\mc{A}$ –∫–∞–∂–¥—ã–π —Ä–∞–∑, –∫–æ–≥–¥–∞ –≥–æ–≤–æ—Ä–∏–º, —á—Ç–æ $A(x) \in \mc{A}$!}.
				\item –í–≤–µ–¥—ë–º –æ—Ç–Ω–æ—à–µ–Ω–∏–µ –Ω–∞ $X\col$ –ø—É—Å—Ç—å $x \sim y,$ –µ—Å–ª–∏ $A(x) = A(y)$. –û—á–µ–≤–∏–¥–Ω–æ, —á—Ç–æ —ç—Ç–æ –æ—Ç–Ω–æ—à–µ–Ω–∏–µ —ç–∫–≤–∏–≤–∞–ª–µ–Ω—Ç–Ω–æ—Å—Ç–∏. –î–æ–∫–∞–∂–µ–º, —á—Ç–æ $x \sim y \eqv y \in A(x)$.

				–ü—É—Å—Ç—å $y \in A(x)$. –ü—Ä–µ–¥–ø–æ–ª–æ–∂–∏–º, —á—Ç–æ $A(y) \neq A(x)$. –¢–æ–≥–¥–∞ –≤—ã–ø–æ–ª–Ω—è–µ—Ç—Å—è –º–∏–Ω–∏–º—É–º –æ–¥–Ω–æ –∏–∑ –¥–≤—É—Ö —É—Ç–≤–µ—Ä–∂–¥–µ–Ω–∏–π: –ª–∏–±–æ $A(y)$ —Å–æ–¥–µ—Ä–∂–∏—Ç —ç–ª–µ–º–µ–Ω—Ç, –∫–æ—Ç–æ—Ä–æ–≥–æ –Ω–µ—Ç –≤ $A(x),$ –ª–∏–±–æ –Ω–∞–æ–±–æ—Ä–æ—Ç. –ü—É—Å—Ç—å –ø–µ—Ä–≤–æ–µ. –¢–æ–≥–¥–∞ $B = A(x) \cap A(y)$~--- —ç–ª–µ–º–µ–Ω—Ç $\mc{A},$ –∫–æ—Ç–æ—Ä—ã–π —Å–æ–¥–µ—Ä–∂–∏—Ç $y$ –∏ —Å—Ç—Ä–æ–≥–æ –º–µ–Ω—å—à–µ $A(y),$ —á–µ–≥–æ –Ω–µ –º–æ–∂–µ—Ç –±—ã—Ç—å. –ü—É—Å—Ç—å –≤—Ç–æ—Ä–æ–µ. –¢–æ–≥–¥–∞ –µ—Å–ª–∏ $A(y)$ –Ω–µ —Å–æ–¥–µ—Ä–∂–∏—Ç $x,$ —Ç–æ $A(x) \setminus A(y)$ —è–≤–ª—è–µ—Ç—Å—è —ç–ª–µ–º–µ–Ω—Ç–æ–º $\mc{A},$ —Å–æ–¥–µ—Ä–∂–∞—â–∏–º $x,$ –∞ –µ—Å–ª–∏ —Å–æ–¥–µ—Ä–∂–∏—Ç, —Ç–æ —Å–Ω–æ–≤–∞ $A(x) \cap A(y)$ —è–≤–ª—è–µ—Ç—Å—è —Ç–∞–∫–∏–º —ç–ª–µ–º–µ–Ω—Ç–æ–º. –ü—Ä–∏—á—ë–º —Å—Ç—Ä–æ–≥–æ –º–µ–Ω—å—à–∏–º, —á–µ–º $A(x),$ —á—Ç–æ –æ–ø—è—Ç—å –≤–µ–¥—ë—Ç –Ω–∞—Å –∫ –ø—Ä–æ—Ç–∏–≤–æ—Ä–µ—á–∏—é.

				–ü—É—Å—Ç—å $A(x) = A(y)$. –ü—Ä–µ–¥–ø–æ–ª–æ–∂–∏–º, —á—Ç–æ $y \notin A(x)$. –ù–æ —Ç–æ–≥–¥–∞ $y \notin A(y),$ —á—Ç–æ —Ç–æ—á–Ω–æ –ª–æ–∂—å.
				\item –†–∞–∑–æ–±—å—ë–º $X$ –Ω–∞ –∫–ª–∞—Å—Å—ã —ç–∫–≤–∏–≤–∞–ª–µ–Ω—Ç–Ω–æ—Å—Ç–∏ –ø–æ –æ—Ç–Ω–æ—à–µ–Ω–∏—é $\sim\scol$ –æ–±–æ–∑–Ω–∞—á–∏–º –º–Ω–æ–∂–µ—Å—Ç–≤–æ —ç—Ç–∏—Ö –∫–ª–∞—Å—Å–æ–≤ $\hat{\mc{A}}$. –ü–æ–Ω—è—Ç–Ω–æ, —á—Ç–æ $|\hat{\mc{A}}| < \infty,$ –≤–µ–¥—å $\hat{\mc{A}} \subset \mc{A}$. –ü—É—Å—Ç—å $B \in \mc{A}$ –∏ $\hat{B} \in \hat{\mc{B}}$. –î–æ–∫–∞–∂–µ–º, —á—Ç–æ –µ—Å–ª–∏ $B \cap \hat{B} \neq \varnothing,$ —Ç–æ $B \cap \hat{B} = \hat{B}$.

				–ü—Ä–µ–¥–ø–æ–ª–æ–∂–∏–º –ø—Ä–æ—Ç–∏–≤–Ω–æ–µ: –ø—É—Å—Ç—å $x \in B \cap \hat{B}$ –∏ $y \in \hat{B} \setminus B$. –ò–∑ –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è –æ—Ç–Ω–æ—à–µ–Ω–∏—è —ç–∫–≤–∏–≤–∞–ª–µ–Ω—Ç–Ω–æ—Å—Ç–∏ –ø–æ–Ω—è—Ç–Ω–æ, —á—Ç–æ $\hat{B} = A(x) = A(y)$. –ù–æ –∑–∞–º–µ—Ç–∏–º —Ç–æ–≥–¥–∞, —á—Ç–æ $\hat{B} \setminus B$~--- –º–Ω–æ–∂–µ—Å—Ç–≤–æ –∏–∑ $\mc{A},$ —Å–æ–¥–µ—Ä–∂–∞—â–µ–µ $y$ –∏ —Å—Ç—Ä–æ–≥–æ –º–µ–Ω—å—à–µ–µ $\hat{B},$ —á–µ–≥–æ –Ω–µ –º–æ–∂–µ—Ç –±—ã—Ç—å.
				\item –ò–∑ —Å–¥–µ–ª–∞–Ω–Ω–æ–≥–æ –Ω–µ—Ç—Ä—É–¥–Ω–æ —É–≤–∏–¥–µ—Ç—å, —á—Ç–æ –ª—é–±–æ–µ $B \in \mc{A}$ –º–æ–∂–Ω–æ –ø—Ä–µ–¥—Å—Ç–∞–≤–∏—Ç—å, –∫–∞–∫ –æ–±—ä–µ–¥–∏–Ω–µ–Ω–∏–µ –º–Ω–æ–∂–µ—Å—Ç–≤ –∏–∑ $\hat{\mc{A}}\col$ –ø—Ä–æ—Å—Ç–æ –¥–ª—è –∫–∞–∂–¥–æ–≥–æ $b \in B$ –≤–∑—è—Ç—å $A(b)$ –∏ –æ–±—ä–µ–¥–∏–Ω–∏—Ç—å –∏—Ö –≤—Å–µ. –ü—Ä–∏ —ç—Ç–æ–º –ø–æ–Ω—è—Ç–Ω–æ, —á—Ç–æ –ª—é–±–æ–µ –æ–±—ä–µ–¥–∏–Ω–µ–Ω–∏–µ –º–Ω–æ–∂–µ—Å—Ç–≤ –∏–∑ $\hat{\mc{A}}$ –ª–µ–∂–∏—Ç –≤ $A$. –¢.–∫. —ç–ª–µ–º–µ–Ω—Ç—ã $\hat{\mc{A}}$ –Ω–µ –ø–µ—Ä–µ—Å–µ–∫–∞—é—Ç—Å—è, –Ω–µ—Ç—Ä—É–¥–Ω–æ —É–≤–∏–¥–µ—Ç—å, —á—Ç–æ –æ—Ç–æ–±—Ä–∞–∂–µ–Ω–∏–µ, —Å–æ–ø–æ—Å—Ç–∞–≤–ª—è—é—â–µ–µ –º–Ω–æ–∂–µ—Å—Ç–≤—É $\mc{B} \subset \hat{\mc{A}}$ –æ–±—ä–µ–¥–∏–Ω–µ–Ω–∏–µ –≤—Å–µ—Ö –µ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–æ–≤ –µ—Å—Ç—å –±–∏–µ–∫—Ü–∏—è~--- –±–∏–µ–∫—Ü–∏—è –º–µ–∂–¥—É –º–Ω–æ–∂–µ—Å—Ç–≤–∞–º–∏ $2^{\hat{\mc{A}}}$ –∏ $\mc{A}$. –ü–æ—ç—Ç–æ–º—É –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —ç–ª–µ–º–µ–Ω—Ç–æ–≤ $\mc{A}$ –∏–º–µ–µ—Ç –∏—Å–∫–æ–º—ã–π –≤–∏–¥.
			\end{enumerate}
		\end{proof}
	\end{exc}

	–ü—Ä–∏–º–µ—Ä—ã –ø—Ä–∏–≤–µ—Å—Ç–∏ –Ω–µ –æ—á–µ–Ω—å —Å–ª–æ–∂–Ω–æ, –Ω–µ –±—É–¥–µ–º –∑–¥–µ—Å—å –Ω–∞ —ç—Ç–æ–º –æ—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞—Ç—å—Å—è.

	\begin{de}
		\ti{$\sigma$-–∞–ª–≥–µ–±—Ä–æ–π} –Ω–∞–∑—ã–≤–∞–µ—Ç—Å—è –∞–ª–≥–µ–±—Ä–∞, –∑–∞–º–∫–Ω—É—Ç–∞—è –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ —Å—á—ë—Ç–Ω—ã—Ö –æ–±—ä–µ–¥–∏–Ω–µ–Ω–∏–π –∏ –ø–µ—Ä–µ—Å–µ—á–µ–Ω–∏–π.
	\end{de}

	\begin{de}
		–ü—É—Å—Ç—å $\mc{E} \subset 2^X.$ –¢–æ–≥–¥–∞ –Ω–∞–∏–º–µ–Ω—å–Ω—à–∞—è $\sigma$-–∞–ª–≥–µ–±—Ä–∞, —Å–æ–¥–µ—Ä–∂–∞—â–∞—è $\mc{E},$ –Ω–∞–∑—ã–≤–∞–µ—Ç—Å—è \ti{–±–æ—Ä–µ–ª–µ–≤—Å–∫–æ–π –æ–±–æ–ª–æ—á–∫–æ–π} $\mc{E}$ –∏ –æ–±–æ–∑–Ω–∞—á–∞–µ—Ç—Å—è $\sigma(\mc{E})$. (–°—Å—ã–ª–∞—è—Å—å –Ω–∞ —Ñ–∞–∫—Ç, –∫–æ—Ç–æ—Ä—ã–π —É–∂–µ —É–ø–æ–º–∏–Ω–∞–ª—Å—è –≤ —É–ø—Ä–∞–∂–Ω–µ–Ω–∏–∏, –∑–∞–º–µ—Ç–∏–º, —á—Ç–æ $\sigma(\mc{E})$ —Å–æ–≤–ø–∞–¥–∞–µ—Ç —Å –ø–µ—Ä–µ—Å–µ—á–µ–Ω–∏–µ–º –≤—Å–µ—Ö $\sigma$-–∞–ª–≥–µ–±—Ä, —Å–æ–¥–µ—Ä–∂–∞—â–∏—Ö $E$).
	\end{de}

	\begin{lm}

\section{–ë–æ—Ä–µ–ª–µ–≤—Å–∫–∞—è \texorpdfstring{$\sigma$}{œÉ}-–∞–ª–≥–µ–±—Ä–∞.}

	\begin{de}

	\begin{thm}
		$\sigma(\Cell_n) = \sigma(\mc{O}_n)$.
		\begin{proof}
			–ó–Ω–∞—è –ø–æ—Å–ª–µ–¥–Ω–∏–π —Ä–µ–∑—É–ª—å—Ç–∞—Ç –∏–∑ –ø—Ä–µ–¥—ã–¥—É—â–µ–≥–æ –±–∏–ª–µ—Ç–∞, –∏–º–µ–µ–º –≤–æ–∑–º–æ–∂–Ω–æ—Å—Ç—å –¥–æ–∫–∞–∑—ã–≤–∞—Ç—å, —á—Ç–æ
			\[
				\Cell_n \subset \sigma(\mc{O}_n) \text{ –∏ } \mc{O}_n \subset
				 \sigma(\Cell_n).
			\]
			–≠—Ç–æ –¥–∞—Å—Ç –Ω–∞–º —É—Ç–≤–µ—Ä–∂–¥–µ–Ω–∏–µ —Ç–µ–æ—Ä–µ–º—ã.

\section{–°–≤–æ–π—Å—Ç–≤–∞ –º–µ—Ä—ã}
	
	\begin{pr}[–ú–æ–Ω–æ—Ç–æ–Ω–Ω–æ—Å—Ç—å]
		–ü—É—Å—Ç—å $A, \, B \in \mc{A}, \; A \subset B$. –¢–æ–≥–¥–∞ $\mu(A) \leqslant \mu(B)$.
		\begin{proof}
			\[

			\]
		\end{proof}
	\end{pr}

	\begin{pr}
		–ü—É—Å—Ç—å $A, \, B \in \mc{A}, \; A \subset B, \; \mu{B} < \infty$. –¢–æ–≥–¥–∞ $\mu(B \setminus A) = \mu{B} - \mu{A}$.
		\begin{proof}
			\[
				\mu(B) = \mu(A) + \mu(B \setminus A) \so \mu(B \setminus A) = \mu(B) - \mu(A).
			\]
			–£—Å–ª–æ–≤–∏–µ $\mu(B) < \infty$ –±—ã–ª–æ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–æ, –∫–æ–≥–¥–∞ –º—ã –≤—ã—á–ª–∏ $\mu(A)$ –∏–∑ –¥–≤—É—Ö —á–∞—Å—Ç–µ–π —Ä–∞–≤–µ–Ω—Å—Ç–≤–∞; –¥–µ–π—Å—Ç–≤–∏—Ç–µ–ª—å–Ω–æ, –ø–æ –ø—Ä–µ–¥—ã–¥—É—â–µ–º—É —Å–≤–æ–π—Å—Ç–≤—É $\mu(A) \leqslant \mu(B) < \infty,$ –ø–æ—ç—Ç–æ–º—É $\mu(A)$ –º–æ–∂–Ω–æ –≤—ã—á–∏—Ç–∞—Ç—å. \footnote{–ù–µ –¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ –ª–∏ –ø–æ—Ç—Ä–µ–±–æ–≤–∞—Ç—å, —á—Ç–æ $\mu(A) < \infty$?}
		\end{proof}
	\end{pr}

	\begin{pr}[–£—Å–∏–ª–µ–Ω–Ω–∞—è –º–æ–Ω–æ—Ç–æ–Ω–Ω–æ—Å—Ç—å]
		–ü—É—Å—Ç—å $A_1, \, ..., \, A_n, \, B \in \mc{A}, \; A_1, \, ..., \, A_n \subset B,$ –ø—Ä–∏—á—ë–º –º–Ω–æ–∂–µ—Å—Ç–≤–∞ $A_k$ –¥–∏–∑—ä—é–Ω–∫—Ç–Ω—ã–µ. –¢–æ–≥–¥–∞ 
		\[
			\sum\limits_{k = 1}^n \mu(A_k) \leqslant \mu(B).
		\] 
		\begin{proof}
			–û—á–µ–≤–∏–¥–Ω–æ.
		\end{proof}
	\end{pr}

	\begin{pr}[–ü–æ–ª—É–∞–¥–¥–∏—Ç–∏–≤–Ω–æ—Å—Ç—å]
		–ü—É—Å—Ç—å $A_1, \, ..., \, A_n, \, B \in \mc{A}, \; B \subset \cup A_k$. –¢–æ–≥–¥–∞
		\[
			\mu(B) \leqslant \sum\limits_{k = 1}^n \mu(A_k).
		\]
		\begin{proof}
			–í–≤–µ–¥—ë–º —Å–µ–º–µ–π—Å—Ç–≤–æ –º–Ω–æ–∂–µ—Å—Ç–≤:
			\[
				C_{k} = A_{k} \setminus \bigcup_{i = 1}^{k - 1} A_i, \; 1 \leqslant k \leqslant n.
			\]
			–ù–µ—Ç—Ä—É–¥–Ω–æ –ø–æ–Ω—è—Ç—å, —á—Ç–æ –æ–Ω–∏ –¥–∏–∑—ä—é–Ω–∫—Ç–Ω—ã; –ø—Ä–∏ —ç—Ç–æ–º 
			\[
				\bigsqcup\limits_{k = 1}^n C_{k} = \cup A_k,
			\]
			–ø–æ—Ç–æ–º—É —á—Ç–æ –Ω–∏–∫–∞–∫–∏—Ö —Ç–æ—á–µ–∫ –∏–∑–≤–Ω–µ $\cup A_k$ –≤ —ç—Ç–æ –æ–±—ä–µ–¥–∏–Ω–µ–Ω–∏–µ —Ç–æ—á–Ω–æ –ø–æ–ø–∞—Å—Ç—å –Ω–µ –º–æ–∂–µ—Ç, –∞ –¥–ª—è –ª—é–±–æ–π —Ç–æ—á–∫–∏ $a$ –∏–∑ $\cup A_k$ –º–æ–∂–Ω–æ –≤–∑—è—Ç—å –Ω–∞–∏–º–µ–Ω—å—à–µ–µ $k_0$ —Ç–∞–∫–æ–µ, —á—Ç–æ $a \in A_{k_0};$ —Ç–æ–≥–¥–∞ $a \in C_{k_0}$. 

			–ò–∑ —ç—Ç–æ–≥–æ —Å–ª–µ–¥—É–µ—Ç, —á—Ç–æ $B$ –º–æ–∂–Ω–æ –ø—Ä–µ–¥—Å—Ç–∞–≤–∏—Ç—å, –∫–∞–∫
			\[
				B = \bigsqcup \limits_{k = 1}^n B \cap C_k = \bigsqcup \limits_{k = 1}^n D_k.
			\]

			–ó–∞–º–µ—Ç–∏–º, —á—Ç–æ 
			\[
				\mu(D_k) = \mu(B \cap C_k) \leqslant \mu(C_k) \leqslant \mu(A_k).
			\]
			–ü–æ—ç—Ç–æ–º—É –∏
			\[
				\mu(B) = \sum\limits_{k = 1}^n \mu(D_k) \leqslant \sum\limits_{k = 1}^n \mu(A_k).
			\]
		\end{proof}
	\end{pr}

	\begin{pr}[–ù–µ–ø—Ä–µ—Ä—ã–≤–Ω–æ—Å—Ç—å –º–µ—Ä—ã —Å–Ω–∏–∑—É]
		–ü—É—Å—Ç—å $\{A_k\}_{k = 1}^{\infty}$~--- —Ç–∞–∫–æ–µ —Å–µ–º–µ–π—Å—Ç–≤–æ –º–Ω–æ–∂–µ—Å—Ç–≤ –∏–∑ $\mc{A},$ —á—Ç–æ $A_k \subset A_{k + 1},$ –∏ 
		\[
			A = \bigcup\limits_{k = 1}^{\infty} A_k \in \mc{A}.
		\]
		–¢–æ–≥–¥–∞ $\mu(A) = \lim \mu(A_k)$.
		\begin{proof}

	\begin{thm}
		–û–±—ä—ë–º~--- —Å—á—ë—Ç–Ω–æ-–∞–¥–¥–∏—Ç–∏–≤–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è –Ω–∞ $\Cell_n$.

				\]

				–° –¥—Ä—É–≥–æ–π —Å—Ç–æ—Ä–æ–Ω—ã,

\section{–°—Ä–∞–≤–Ω–µ–Ω–∏–µ –∏–Ω—Ç–µ–≥—Ä–∞–ª–∞ –ø–æ –º–µ—Ä–µ —Å –Ω–µ—Å–æ–±—Å—Ç–≤–µ–Ω–Ω—ã–º –∏–Ω—Ç–µ–≥—Ä–∞–ª–æ–º}

	\begin{de}[–ù–∞–ø–æ–º–∏–Ω–∞–Ω–∏–µ]
		–ü—É—Å—Ç—å $f$ –Ω–µ–ø—Ä–µ—Ä—ã–≤–Ω–∞ –Ω–∞ $[a, \, b)$. –¢–æ–≥–¥–∞ \ti{–Ω–µ—Å–æ–±—Å—Ç–≤–µ–Ω–Ω—ã–π –∏–Ω—Ç–µ–≥—Ä–∞–ª} –ø–æ —ç—Ç–æ–º—É –ø—Ä–æ–º–µ–∂—É—Ç–∫—É~---
		\[

		\]
	\end{de}

	\begin{thm}
		–ü—É—Å—Ç—å –Ω–µ–ø—Ä–µ—Ä—ã–≤–Ω–∞—è $f$ –ª–∏–±–æ –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω–∞, –ª–∏–±–æ —Å—É–º–º–∏—Ä—É–µ–º–∞ –Ω–∞ $[a, \, b),$ —Ç–æ–≥–¥–∞
		\[
			\int\limits_{[a, \, b)} f\D \lambda = \int\limits_a^{\to b} f.
		\]
		\begin{proof}
			–†–∞—Å—Å–º–æ—Ç—Ä–∏–º 
			\[
				F(x) = \int\limits_{[a, \, x]} f.
			\]
			–ü–æ–Ω—è—Ç–Ω–æ, —á—Ç–æ\footnote{–ï—Å–ª–∏ —ç—Ç–æ—Ç –ø—Ä–µ–¥–µ–ª —Å—É—â–µ—Å—Ç–≤—É–µ—Ç.}
			\[
				\lim\limits_{x \to b} F(x) = \int\limits_a^{\to b} f,
			\]
			–ø–æ—Ç–æ–º—É —á—Ç–æ –º—ã —É–∂–µ –¥–æ–∫–∞–∑–∞–ª–∏, —á—Ç–æ –∏–Ω—Ç–µ–≥—Ä–∞–ª—ã –ø–æ –æ—Ç—Ä–µ–∑–∫—É –æ—Ç –Ω–µ–ø—Ä–µ—Ä—ã–≤–Ω–æ–π —Ñ—É–Ω–∫—Ü–∏–∏ –ø–æ –õ–µ–±–µ–≥—É –∏ –ø–æ –†–∏–º–∞–Ω—É —Å–æ–≤–ø–∞–¥–∞—é—Ç.

			–ù—É–∂–Ω–æ –¥–æ–∫–∞–∑–∞—Ç—å, —á—Ç–æ
			\[
				\lim\limits_{x \to b} F(x) = \int\limits_{[a, \, b)} f.
			\]
			–ï—Å–ª–∏ $f$ —Å—É–º–º–∏—Ä—É–µ–º–∞, —Ç–æ —ç—Ç–æ —Å–ª–µ–¥—É–µ—Ç –∏–∑ —Ç–µ–æ—Ä–µ–º—ã –æ–± –∞–±—Å–æ–ª—é—Ç–Ω–æ–π –Ω–µ–ø—Ä–µ—Ä—ã–≤–Ω–æ—Å—Ç–∏ –∏–Ω—Ç–µ–≥—Ä–∞–ª–∞ –õ–µ–±–µ–≥–∞ (–∏ —Å—É—â–µ—Å—Ç–≤–æ–≤–∞–Ω–∏–µ –ø—Ä–µ–¥–µ–ª–∞ –æ–∫–∞–∑—ã–≤–∞–µ—Ç—Å—è —Å–æ–≤—Å–µ–º –æ—á–µ–≤–∏–¥–Ω—ã–º).

			–†–∞—Å—Å–º–æ—Ç—Ä–∏–º —Å–ª—É—á–∞–π, –∫–æ–≥–¥–∞ $f$ –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω–∞, –Ω–æ –Ω–µ —Å—É–º–º–∏—Ä—É–µ–º–∞. –ú—ã –∑–Ω–∞–µ–º, —á—Ç–æ –∏–Ω—Ç–µ–≥—Ä–∞–ª –õ–µ–±–µ–≥–∞ –æ—Ç –Ω–µ—ë –ø–æ $[a, \, b)$ —Ä–∞–≤–µ–Ω $+\infty,$ –∏ –Ω—É–∂–Ω–æ –ª–∏—à—å –¥–æ–∫–∞–∑–∞—Ç—å, —á—Ç–æ –ø—Ä–µ–¥–µ–ª $F(x)$ —Å—É—â–µ—Å—Ç–≤—É–µ—Ç –∏ –Ω–µ –º–æ–∂–µ—Ç –±—ã—Ç—å –∫–æ–Ω–µ—á–µ–Ω. –°—É—â–µ—Å—Ç–≤—É–µ—Ç –æ–Ω –ø–æ—Ç–æ–º—É, —á—Ç–æ $F(x)$ –±—É–¥–µ—Ç —Ñ—É–Ω–∫—Ü–∏–µ–π –≤–æ–∑—Ä–∞—Å—Ç–∞—é—â–µ–π. –ö–æ–Ω–µ—á–Ω—ã–º –∂–µ –æ–Ω –±—ã—Ç—å –Ω–µ –º–æ–∂–µ—Ç, –ø–æ—Ç–æ–º—É —á—Ç–æ —ç—Ç–æ –ø—Ä–æ—Ç–∏–≤–æ—Ä–µ—á–∏–ª–æ –±—ã —Ç–µ–æ—Ä–µ–º–µ –õ–µ–≤–∏, —á—Ç–æ —Å–µ–π—á–∞—Å –∏ –ø–æ–∫–∞–∂–µ–º.

			–í–æ–∑—å–º—ë–º –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—å —Ç–æ—á–µ–∫ $\{x_n\}_{n = 1}^{\infty}$ –∏–∑ $[a, \, b),$ —Å—Ö–æ–¥—è—â—É—é—Å—è –∫ $b$. –ó–∞–º–µ—Ç–∏–º, —á—Ç–æ

			–§—É–Ω–∫—Ü–∏–∏ $f \, \mathbb{1}_{[a, \, x_n]}$ –æ–±—Ä–∞–∑—É—é—Ç –Ω–µ—É–±—ã–≤—é—â—É—é –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—É—é –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—å, —Å—Ö–æ–¥—è—â—É—é—Å—è –∫ $f,$ –ø–æ—ç—Ç–æ–º—É –ø–æ —Ç–µ–æ—Ä–µ–º–µ –õ–µ–≤–∏ —ç—Ç–æ—Ç –ø—Ä–µ–¥–µ–ª –±—É–¥–µ—Ç —Ä–∞–≤–µ–Ω –∫–∞–∫ —Ä–∞–∑ $+\infty$.
		\end{proof}
	\end{thm}

	\begin{exm}
		–£—Å–ª–æ–≤–Ω–æ —Å—Ö–æ–¥—è—â–∏–π—Å—è –∏–Ω—Ç–µ–≥—Ä–∞–ª
		\[

\section{–ò–Ω—Ç–µ–≥—Ä–∞–ª –ø–æ –¥–∏—Å–∫—Ä–µ—Ç–Ω–æ–π –º–µ—Ä–µ –∏ –ø–æ –º–µ—Ä–µ, –∑–∞–¥–∞–≤–∞–µ–º–æ–π –ø–ª–æ—Ç–Ω–æ—Å—Ç—å—é}

	\begin{de}
		–ü—É—Å—Ç—å $X$~--- –º–Ω–æ–∂–µ—Å—Ç–≤–æ, $\mc{A} = 2^{X}$ –∏ –µ—Å—Ç—å –Ω–µ –±–æ–ª–µ–µ —á–µ–º —Å—á—ë—Ç–Ω—ã–µ –º–Ω–æ–∂–µ—Å—Ç–≤–∞ $\{a_i\} \in X$ –∏ $\{m_i\} \in \R$. –¢–æ–≥–¥–∞ \ti{–¥–∏—Å–∫—Ä–µ—Ç–Ω–∞—è –º–µ—Ä–∞} –∑–∞–¥–∞—ë—Ç—Å—è, –∫–∞–∫
		\[
			\mu(E) = \sum\limits_{i} m_i \delta_{a_i}(E).
		\]
	\end{de}

	\begin{lm}
		–ò–Ω—Ç–µ–≥—Ä–∞–ª –æ—Ç –ª—é–±–æ–π (–∏–∑–º–µ—Ä–∏–º–æ–π) —Ñ—É–Ω–∫—Ü–∏–∏ –ø–æ –º–Ω–æ–∂–µ—Å—Ç–≤—É $E$ –Ω—É–ª–µ–≤–æ–π –º–µ—Ä—ã —Ä–∞–≤–µ–Ω –Ω—É–ª—é.
		\begin{proof}
			–î–ª—è –Ω–∞—á–∞–ª–∞ –º–æ–∂–Ω–æ –∑–∞–º–µ—Ç–∏—Ç—å, —á—Ç–æ –¥–ª—è –ø—Ä–æ—Å—Ç—ã—Ö —Ñ—É–Ω–∫—Ü–∏–π —ç—Ç–æ —Ç–æ—á–Ω–æ —Ç–∞–∫. –î–µ–π—Å—Ç–≤–∏—Ç–µ–ª—å–Ω–æ, –ø—É—Å—Ç—å
			\[

			\]
			–¢–æ–≥–¥–∞
			\[
				\int\limits_{E} \sum\limits_{k = 1}^p a_k \mathbb{1}_{E_k} = \sum\limits_{k = 1}^p a_k \int\limits_{X} \underbrace{\mathbb{1}_{E_k \cap E}}_{=0} = 0.
			\]

			–ù–æ —Ç–æ–≥–¥–∞ –ø–æ–Ω—è—Ç–Ω–æ, —á—Ç–æ –¥–ª—è –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã—Ö —Ñ—É–Ω–∫—Ü–∏–π —ç—Ç–æ —Ç–æ–∂–µ –±—É–¥–µ—Ç –≤–µ—Ä–Ω–æ, –ø–æ—Ç–æ–º—É —á—Ç–æ —Å—É–ø—Ä–µ–º—É–º –Ω—É–ª–µ–π –Ω–æ–ª—å. –ï—â—ë –±–æ–ª–µ–µ –æ—á–µ–≤–∏–¥–Ω–æ, —á—Ç–æ –¥–ª—è –ø—Ä–æ–∏–∑–≤–æ–ª—å–Ω—ã—Ö –∏–∑–º–µ—Ä–∏–º—ã—Ö —Ñ—É–Ω–∫—Ü–∏–π –Ω–∏—á–µ–≥–æ –Ω–µ –∏–∑–º–µ–Ω–∏—Ç—Å—è.

	\end{lm}

	\begin{lm}
		–ò–Ω—Ç–µ–≥—Ä–∞–ª –æ—Ç –∏–∑–º–µ—Ä–∏–º–æ–π —Ñ—É–Ω–∫—Ü–∏–∏ $f$ –ø–æ –º–Ω–æ–∂–µ—Å—Ç–≤—É ${a}$ —Ä–∞–≤–µ–Ω $f(a) \mu\big(\{a\}\big)$.

	\end{lm}

	\begin{thm}
		–ü—É—Å—Ç—å $f\col \; X \to \R$ –ª–∏–±–æ –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω–∞, –ª–∏–±–æ —Å—É–º–º–∏—Ä—É–µ–º–∞ –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ –¥–∏—Å–∫—Ä–µ—Ç–Ω–æ–π –º–µ—Ä—ã. –¢–æ–≥–¥–∞

		\]

		\end{proof}
	\end{thm}

	\begin{st}
		–î–ª—è –¥–∏—Å–∫—Ä–µ—Ç–Ω–æ–π –º–µ—Ä—ã —Å—É–º–º–∏—Ä—É–µ–º–æ—Å—Ç—å —Ñ—É–Ω–∫—Ü–∏–∏ —Ä–∞–≤–Ω–æ—Å–∏–ª—å–Ω–∞ –∞–±—Å–æ–ª—é—Ç–Ω–æ–π —Å—Ö–æ–¥–∏–º–æ—Å—Ç–∏ —Ä—è–¥–∞, –∑–∞–ø–∏—Å–∞–Ω–Ω–æ–≥–æ –≤ –ø—Ä–µ–¥—ã–¥—É—â–µ–π —Ç–µ–æ—Ä–µ–º–µ.
		\begin{proof}
			–°—É–º–º–∏—Ä—É–µ–º–æ—Å—Ç—å —Ñ—É–Ω–∫—Ü–∏–∏ —Ä–∞–≤–Ω–æ—Å–∏–ª—å–Ω–∞ —Å—É–º–º–∏—Ä—É–µ–º–æ—Å—Ç–∏ –µ—ë –º–æ–¥—É–ª—è. –ú–æ–¥—É–ª—å –∂–µ —Ñ—É–Ω–∫—Ü–∏—è –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω–∞—è, –¥–ª—è –Ω–µ–≥–æ –≤—ã–ø–æ–ª–Ω—è–µ—Ç—Å—è –ø—Ä–µ–¥—ã–¥—É—â–∞—è —Ç–µ–æ—Ä–µ–º–∞, –∏ –∏–Ω—Ç–µ–≥—Ä–∞–ª –æ—Ç –Ω–µ–≥–æ —Ä–∞–≤–µ–Ω —Å—É–º–º–µ –∏–∑ –º–æ–¥—É–ª–µ–π. –ó–Ω–∞—á–∏—Ç, –∏ –∏—Ö —Å—Ö–æ–¥–∏–º–æ—Å—Ç–∏ —Ä–∞–≤–Ω–æ—Å–∏–ª—å–Ω—ã.
		\end{proof}
	\end{st}

	\begin{exm}
		–ï—Å–ª–∏, –Ω–∞–ø—Ä–∏–º–µ—Ä, –≤–∑—è—Ç—å $X = \N, \; a_k = k$ –∏ $m_k = 1,$ —Ç–æ –º–µ—Ä–∞ –±—É–¥–µ—Ç –æ–±–æ–∑–Ω–∞—á–∞—Ç—å –ø—Ä–æ—Å—Ç–æ —Å—É–º–º—É –∑–Ω–∞—á–µ–Ω–∏–π —Ñ—É–Ω–∫—Ü–∏–∏ –≤ —Ç–æ—á–∫–∞—Ö. –§—É–Ω–∫—Ü–∏—è –∏–∑ $\N$ –≤ $\R$~--- —Ä—è–¥, –∞ —Å—É–º–º–∏—Ä—É–µ–º–æ—Å—Ç—å~--- –∞–±—Å–æ–ª—é—Ç–Ω–∞—è —Å—Ö–æ–¥–∏–º–æ—Å—Ç—å. 
	\end{exm}

	\begin{de}
		–ü—É—Å—Ç—å $X$~--- –ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–æ —Å –º–µ—Ä–æ–π $\mu,$ –∏ –µ—Å—Ç—å –∏–∑–º–µ—Ä–∏–º–∞—è –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è $\rho\col \; X \to \R$. –¢–æ–≥–¥–∞ –º–æ–∂–Ω–æ –≤–≤–µ—Å—Ç–∏ –º–µ—Ä—É
		\[
			\nu(E) = \int\limits_{E} \rho \D \mu.
		\]
	\end{de}

	\begin{st}
		$\nu$ –∏ –ø—Ä–∞–≤–¥–∞ –º–µ—Ä–∞.
		\begin{proof}
			–ü–µ—Ä–≤–∞—è –∞–∫—Å–∏–æ–º–∞ –æ—á–µ–≤–∏–¥–Ω–∞, –∞ –≤—Ç–æ—Ä–∞—è —Å–ª–µ–¥—É–µ—Ç –∏–∑ —Ç–µ–æ—Ä–µ–º—ã –æ —Å—á—ë—Ç–Ω–æ–π –∞–¥–¥–∏—Ç–∏–≤–Ω–æ—Å—Ç–∏ –∏–Ω—Ç–µ–≥—Ä–∞–ª–∞~--- –≤–µ–¥—å –Ω–∞—à–∞ $\rho$ –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω–∞.
		\end{proof}
	\end{st}

	\begin{thm}
		–ü—É—Å—Ç—å $f$ –∏–∑–º–µ—Ä–∏–º–∞ –Ω–∞ $X$ –∏ –ª–∏–±–æ –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω–∞, –ª–∏–±–æ —Å—É–º–º–∏—Ä—É–µ–º–∞ –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ $\nu$. –¢–æ–≥–¥–∞
		\[
			\int\limits_{X} f \D \nu = \int\limits_{X} f \rho \D \mu. 
		\]
		\begin{proof}
			$\hphantom{.}$
			\begin{enumerate}
				\item –ü—É—Å—Ç—å —Å–Ω–∞—á–∞–ª–∞ $f$~--- –ø—Ä–æ—Å—Ç–∞—è:
				\[
					f = \sum\limits_{k = 1}^p a_k \mathbb{1}_{E_k}.
				\]
				–¢–æ–≥–¥–∞
				\[
					\int\limits_{X} f \D \nu = \sum\limits_{k = 1}^p a_k \nu(E_k) = \sum\limits_{k = 1}^p a_k \int\limits_{E_k} \rho \D \mu = \sum\limits_{k = 1}^p a_k \int\limits_{X} \mathbb{1}_{E_k} \, \rho \D \mu = \int\limits_{X} f \rho \D \mu.
				\]
				\item –ï—Å–ª–∏ $f$~--- –∏–∑–º–µ—Ä–∏–º–∞—è –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω–∞—è, –º–æ–∂–Ω–æ –≤—ã–¥–µ–ª–∏—Ç—å –Ω–µ—É–±—ã–≤–∞—é—â—É—é –Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—É—é –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—å –ø—Ä–æ—Å—Ç—ã—Ö, —Å—Ö–æ–¥—è—â—É—é—Å—è –∫ –Ω–µ–π. –û—Ç —É–º–Ω–æ–∂–µ–Ω–∏—è –Ω–∞ $g$ –æ–Ω–∞ —ç—Ç–∏—Ö —Å–≤–æ–π—Å—Ç–≤ –Ω–µ –ø–æ—Ç–µ—Ä—è–µ—Ç, –ø–æ—ç—Ç–æ–º—É —Ä–∞–≤–µ–Ω—Å—Ç–≤–æ –±–ª–∞–≥–æ–ø–æ–ª—É—á–Ω–æ –ø–µ—Ä–µ–Ω–µ—Å—ë—Ç—Å—è –ø–æ —Ç–µ–æ—Ä–µ–º–µ –õ–µ–≤–∏.
				\item –ù—É –∏ –¥–ª—è –ø—Ä–æ–∏–∑–≤–æ–ª—å–Ω–æ–π —Å—É–º–º–∏—Ä—É–µ–º–æ–π –Ω—É–∂–Ω–æ –ø—Ä–æ—Å—Ç–æ –Ω–∞–ø–∏—Å–∞—Ç—å.
			\end{enumerate}
		\end{proof}
	\end{thm}
